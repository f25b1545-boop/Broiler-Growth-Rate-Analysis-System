    <script>
        // Default Constants for Cobb 500 Broilers under normal Malaysian climate parameters
        let base_A = 3200; // Asymptotic maximum weight in grams under optimal conditions
        let base_B = 4.33; // Biological scaling constant based on ~42g hatchling weight
        let base_k = 0.0666; // Growth velocity coefficient

        // Elements
        const heatStressSlider = document.getElementById('heat-stress');
        const heatStressVal = document.getElementById('heat-stress-val');
        const nutritionSlider = document.getElementById('nutrition');
        const nutritionVal = document.getElementById('nutrition-val');
        
        // Report Elements
        const reportDay = document.getElementById('report-day');
        const reportDayCircle = document.getElementById('report-day-circle');
        const calcWeight = document.getElementById('calc-weight');
        const calcAdg = document.getElementById('calc-adg');
        const laymanExplanation = document.getElementById('layman-explanation');

        // Global Chart Instance
        let growthChart;
        let selectedDay = 21; // Default selected point

        // Calculations
        function getParameters() {
            // Apply Heat Stress correction to Growth Coefficient k (highly susceptible in Malaysia's tropical humidity)
            const temperature = parseFloat(heatStressSlider.value);
            // Cobb 500 thermoneutral zone is 20-24C, performance declines as temperature exceeds 30C
            let tempFactor = 1.0;
            if (temperature > 30) {
                tempFactor = 1.0 - (temperature - 30) * 0.035; // 3.5% decay per degree above 30C
            } else if (temperature < 30) {
                tempFactor = 1.0 + (30 - temperature) * 0.01; // Tiny gain at cool optimum house temp
            }
            const k = base_k * tempFactor;

            // Apply Protein/Nutrition adjustment to the asymptotic final weight (A)
            const nutritionPercent = parseFloat(nutritionSlider.value) / 100;
            const A = base_A * nutritionPercent;

            return { A, B: base_B, k };
        }

        // Gompertz Function f(t)
        function f(t, A, B, k) {
            return A * Math.exp(-B * Math.exp(-k * t));
        }

        // First Derivative Function f'(t)
        function df(t, A, B, k) {
            const fVal = f(t, A, B, k);
            return k * B * Math.exp(-k * t) * fVal;
        }

        // Generate data arrays for Chart.js
        function generateChartData() {
            const labels = [];
            const weights = [];
            const growthRates = [];
            const { A, B, k } = getParameters();

            for (let day = 0; day <= 42; day++) {
                labels.push(day);
                weights.push(parseFloat(f(day, A, B, k).toFixed(1)));
                growthRates.push(parseFloat(df(day, A, B, k).toFixed(2)));
            }

            return { labels, weights, growthRates };
        }

        // Generate Layman Explanation for Selected Day
        function generateLaymanExplanation(day, weight, adg, k, A) {
            const { B } = getParameters();
            // Calculate mathematical inflection point: t_inflection = ln(B) / k
            const inflectionDay = Math.round(Math.log(B) / k);

            let text = "";

            if (day <= 7) {
                text = `<strong>[Starter Phase - Day ${day}]:</strong> Broilers are in the early brooding stage. Currently, the daily growth rate is only <strong>${adg} g/day</strong>. While this seems low, it represents rapid relative cellular division. Focus on keeping the floor warm, managing high-quality starter protein crumble, and encouraging feed/water access. This is the foundation of structural organ development.`;
            } else if (day > 7 && day < inflectionDay) {
                text = `<strong>[Early Grower Phase - Day ${day}]:</strong> Your flock is accelerating rapidly. The daily gain has hit <strong>${adg} g/day</strong> and is climbing daily as the derivative curve rises. This is the maximum bone development window. Ensure house ventilation is pristine to prevent ammonia buildup.`;
            } else if (day === inflectionDay || Math.abs(day - inflectionDay) <= 1) {
                text = `<strong>[Peak Inflection Point - Day ${day}]:</strong> Crucial moment! The growth rate has peaked at <strong>${adg} g/day</strong>. Mathematically, the second derivative of growth is zero (\(f''(t) = 0\)). This is the point of maximum metabolic efficiency. Ensure zero thermal shock to preserve feed conversion ratio (FCR). Feed should now transition to high-density Grower to leverage this peak capacity.`;
            } else if (day > inflectionDay && day <= 35) {
                text = `<strong>[Finisher Phase - Day ${day}]:</strong> The growth rate (derivative) is decelerating (down to <strong>${adg} g/day</strong>) even though cumulative weight is still rising to <strong>${weight}g</strong>. The birds are storing more fat now than muscle tissue. This is the prime market-readiness target window used by leading poultry integrators in Malaysia to schedule transport.`;
            } else {
                text = `<strong>[Extended Finisher Phase - Day ${day}]:</strong> Over-mature phase. Growth velocity has slumped to <strong>${adg} g/day</strong>. Holding the flock past this day consumes excessive finisher feed for minimal muscle returns. Feed conversion ratio (FCR) is rapidly deteriorating; immediate processing plant dispatch is strongly recommended to protect profit margins.`;
            }

            return text;
        }

        // Update calculations and update the visual report card
        function updateReport(day) {
            selectedDay = day;
            const { A, B, k } = getParameters();
            const weightVal = f(day, A, B, k);
            const adgVal = df(day, A, B, k);

            reportDay.innerText = day;
            reportDayCircle.innerText = day;
            calcWeight.innerText = `${Math.round(weightVal).toLocaleString()} g`;
            calcAdg.innerText = `${adgVal.toFixed(1)} g/day`;

            laymanExplanation.innerHTML = generateLaymanExplanation(day, Math.round(weightVal), adgVal.toFixed(1), k, A);
        }

        // Initialize and setup the Chart
        function initChart() {
            const ctx = document.getElementById('growthChart').getContext('2d');
            const data = generateChartData();

            growthChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: data.labels,
                    datasets: [
                        {
                            label: 'Live Weight f(t) (g)',
                            data: data.weights,
                            borderColor: '#38bdf8', // Tailwind sky-400
                            backgroundColor: 'rgba(56, 189, 248, 0.08)',
                            borderWidth: 3,
                            yAxisID: 'yWeight',
                            pointRadius: (ctx) => ctx.dataIndex === selectedDay ? 8 : 2,
                            pointBackgroundColor: (ctx) => ctx.dataIndex === selectedDay ? '#ffffff' : '#38bdf8',
                            pointBorderColor: '#0ea5e9',
                            pointHoverRadius: 9,
                            tension: 0.1,
                            fill: true
                        },
                        {
                            label: "Daily Growth Rate f'(t) (g/day)",
                            data: data.growthRates,
                            borderColor: '#34d399', // Tailwind emerald-400
                            backgroundColor: 'rgba(52, 211, 153, 0.03)',
                            borderWidth: 2.5,
                            borderDash: [5, 5],
                            yAxisID: 'yRate',
                            pointRadius: (ctx) => ctx.dataIndex === selectedDay ? 8 : 0,
                            pointBackgroundColor: (ctx) => ctx.dataIndex === selectedDay ? '#34d399' : 'transparent',
                            pointHoverRadius: 9,
                            tension: 0.2
                        }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    onClick: (event, activeElements) => {
                        if (activeElements.length > 0) {
                            const index = activeElements[0].index;
                            updateReport(index);
                            growthChart.update();
                        }
                    },
                    interaction: {
                        mode: 'index',
                        intersect: false
                    },
                    plugins: {
                        legend: {
                            display: false // Using custom styled legend in UI
                        },
                        tooltip: {
                            backgroundColor: '#0f172a',
                            titleColor: '#f8fafc',
                            bodyColor: '#cbd5e1',
                            borderColor: '#334155',
                            borderWidth: 1,
                            padding: 10,
                            callbacks: {
                                title: (items) => `Day ${items[0].label} of Grow-out`,
                                label: (context) => {
                                    if (context.datasetIndex === 0) {
                                        return `Weight f(t): ${context.parsed.y} grams`;
                                    } else {
                                        return `Gain Rate f'(t): ${context.parsed.y} g/day`;
                                    }
                                }
                            }
                        }
                    },
                    scales: {
                        x: {
                            grid: { color: 'rgba(51, 65, 85, 0.3)' },
                            ticks: { color: '#94a3b8', font: { family: 'Plus Jakarta Sans' } },
                            title: {
                                display: true,
                                text: 'Age (Days since Hatching)',
                                color: '#94a3b8',
                                font: { size: 11, weight: 'bold', family: 'Plus Jakarta Sans' }
                            }
                        },
                        yWeight: {
                            type: 'linear',
                            position: 'left',
                            grid: { color: 'rgba(51, 65, 85, 0.3)' },
                            ticks: { color: '#38bdf8' },
                            title: {
                                display: true,
                                text: 'Cumulative Body Weight f(t) (grams)',
                                color: '#38bdf8',
                                font: { size: 11, weight: 'bold', family: 'Plus Jakarta Sans' }
                            }
                        },
                        yRate: {
                            type: 'linear',
                            position: 'right',
                            grid: { drawOnChartArea: false }, // avoid grid overlaps
                            ticks: { color: '#34d399' },
                            title: {
                                display: true,
                                text: "Daily Gain f'(t) (g/day)",
                                color: '#34d399',
                                font: { size: 11, weight: 'bold', family: 'Plus Jakarta Sans' }
                            }
                        }
                    }
                }
            });
        }

        // Live calculation triggers when sliders are used
        function handleControls() {
            const tempVal = parseFloat(heatStressSlider.value);
            heatStressVal.innerText = `${tempVal}°C (${tempVal > 30 ? 'High Heat Stress' : 'Thermally Controlled'})`;

            const nutrVal = parseFloat(nutritionSlider.value);
            let nutrDesc = "Optimal feed mix";
            if (nutrVal < 90) nutrDesc = "Deficient amino-acids/protein";
            else if (nutrVal > 110) nutrDesc = "Super-fortified premium feed";
            nutritionVal.innerText = nutrDesc;

            // Regenerate values & animate chart update
            const data = generateChartData();
            growthChart.data.datasets[0].data = data.weights;
            growthChart.data.datasets[1].data = data.growthRates;
            
            updateReport(selectedDay);
            growthChart.update();
        }

        // Wire Event Listeners
        heatStressSlider.addEventListener('input', handleControls);
        nutritionSlider.addEventListener('input', handleControls);

        // Initialize application on window load
        window.onload = function() {
            initChart();
            updateReport(21);
        };
    </script>
